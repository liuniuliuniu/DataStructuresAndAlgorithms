## 反转链表

### Question
输入一个链表，反转链表后，输出新链表的表头。


### Solution


```
class Solution {
    func reverseList(_ head: ListNode?) -> ListNode? {
        var pre: ListNode? = nil
        var current = head
        while current != nil {
            let next = current?.next
            current?.next = pre
            pre = current
            current = next
        }
        return pre
    }
}


```
